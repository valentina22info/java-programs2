import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int dig = sc.nextInt();
        int count = 0 , rem;
        while(num != 0){
            rem = num % 10;
            if(rem == dig){
                count++;
            }
        num = num / 10;
            }
        System.out.println(count);
        }
    }
