import java.util.*;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num = sc.nextInt();
        int original = num;
        int temp = num;
        int digits = 0;
        while (temp > 0) {
            digits++;
            temp = temp / 10;
        }
        temp = num;
        int sum = 0;
        while (temp > 0) {
            int digit = temp % 10;
            int power = 1;
            int count = digits;
            while (count > 0) {
                power = power * digit;
                count--;
            }
            sum = sum + power;
            digits--;      
            temp = temp / 10;
        }
        if (sum == original)
            System.out.println("Yes");
        else
            System.out.println("No");
    }
}
