import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int product = 1 , rem;
        while (a != 0)
        {
            rem = a % 10;
            product = product * rem;
            a = a / 10;
        }
        System.out.println(product);
    }
}
